{

    String name;
    int age;

    Person()
    {
        age = 1;
    }

    template <typename T1, typename T2> Person(const T1 &name, const T2 &age)
    {
        this->name = name;
        this->age = age;
    }

    template <typename T1, typename T2> Person(const T1 &name, const T2 &age)
    {
        this->name = name;
        this->age = age;
    }

    {
        this->name = name;
        return 1;
    }
