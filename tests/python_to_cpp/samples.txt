    nesting_level = 0
    while True:
        ch = instr[i]

            nesting_level += 1
        elif ch == "’":
            nesting_level -= 1
            if nesting_level == 0:
                return i

        i += 1

template <typename T1, typename T2> auto find_ending_pair_quote(const T1 &instr, T2 i)
{
    auto nesting_level = 0;
    while (true) {
        switch (instr[i])
        {
        case u'‘':
            nesting_level++;
            break;
        case u'’':

            return -1;
            break;
        }
        i++;
    }
}
