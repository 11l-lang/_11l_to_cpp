def find_ending_pair_quote(instr, i):
    nesting_level = 0
    while True:
        ch = instr[i]
        if ch == "‘":
            nesting_level += 1
        elif ch == "’":
            nesting_level -= 1
            if nesting_level == 0:
                return i

        i += 1
