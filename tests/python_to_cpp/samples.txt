template <typename T1> auto parse_int(const T1 &s)
{
    auto i = 0;
    auto r = 0;
    while (i < s.len() && in(s[i], range_ee(u'0'_C, u'9'_C))) {

template <typename T1, typename T2> auto find_ending_pair_quote(const T1 &instr, T2 i)
{
    auto nesting_level = 0;
    while (true) {
        switch (instr[i])
        {
        case u'‘':
            nesting_level++;
            break;
        case u'’':
            if (--nesting_level == 0)
                return i;
            break;
        default:
            return -1;
            break;
        }
        i++;
    }
}

{

        if (i == instr.len())
